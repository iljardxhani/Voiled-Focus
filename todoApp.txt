document.addEventListener('DOMContentLoaded', () => {
  const addTaskBtn = document.getElementById('addTaskBtn');
  const taskInput = document.getElementById('taskInput');
  const taskList = document.getElementById('taskList');
  let focusedTime = 0; // Initialize the focused time tracker

  // Add task functionality
  addTaskBtn.addEventListener('click', () => {
    const taskText = taskInput.value.trim();
    if (taskText !== '') {
      const taskItem = document.createElement('li');
      taskItem.classList.add('task');

      taskItem.innerHTML = `
        <div class="task-text">
          <span>${taskText}</span>
        </div>
        <button class="removeBtn">X</button>
        <button class="playBtn">â–¶</button>
        <div class="progress-container">
          <div class="progress-bar green-bar"></div>
        </div>
        <div class="status-sign">Working</div>
      `;

      // Add event listener to remove task
      const removeBtn = taskItem.querySelector('.removeBtn');
      removeBtn.addEventListener('click', () => {
        taskItem.remove();
      });

      // Add event listener to play button for loading effect
      const playBtn = taskItem.querySelector('.playBtn');
      const progressBar = taskItem.querySelector('.progress-bar');
      const statusSign = taskItem.querySelector('.status-sign');
      playBtn.addEventListener('click', () => {
        startTask(playBtn, progressBar, taskItem, statusSign); // Pass taskItem to startTask
      });

      // Append new task to the list
      taskList.appendChild(taskItem);
      taskInput.value = ''; // Clear input field
    }
  });

  // Function to start task, simulate loading effect on progress bar inside task container
  function startTask(playBtn, progressBar, taskContainer, statusSign) {
    let progress = 0;
    const totalTime = 100; // Total time in percentage (0-100)
    playBtn.disabled = true; // Disable play button during task
    playBtn.classList.add('hidden'); // Hide play button
    taskContainer.querySelector('.removeBtn').classList.add('hidden'); // Hide remove button
    statusSign.style.display = 'inline-block'; // Show the "Working" status

    // Set interval to fill the green progress bar 1% at a time
    const interval = setInterval(() => {
      progress += 1; // Increase progress by 1% each time
      progressBar.style.width = `${progress}%`; // Update progress width

      // Once progress reaches 100%, change status to "Coffee Break" and start blue progress bar
      if (progress >= totalTime) {
        clearInterval(interval); // Stop the progress
        focusedTime += 1; // Increase by 1, or whatever logic you need

        // Update the .progress div with the new focused time
        const progressDiv = document.querySelector('.progress'); // Select the progress div
        progressDiv.textContent = `Focused: ${focusedTime}`; // Update the displayed value
        progressBar.classList.remove('green-bar');
        progressBar.classList.add('transparent-bar');

        statusSign.textContent = 'Coffee Break'; // Change status to "Coffee Break"
        startCoffeeBreak(taskContainer, statusSign); // Start the second (blue) progress bar
      }
    }, 20); // Adjust this value to change the speed of the progress bar filling
  }

  // Function to start the Coffee Break progress
  function startCoffeeBreak(taskContainer, statusSign) {
    let progress = 0;
    const coffeeBreakProgressBar = document.createElement('div');
    coffeeBreakProgressBar.classList.add('progress-bar', 'blue-bar'); // Create blue progress bar
    taskContainer.querySelector('.progress-container').appendChild(coffeeBreakProgressBar); // Add to the container

    // Using requestAnimationFrame for smooth animation instead of setInterval
    function updateProgress() {
      const taskItem = taskItem.querySelector(".task-text")

      
      if (progress < 100) {
        progress += 1; // Increase progress by 1% each time
        coffeeBreakProgressBar.style.width = `${progress}%`; // Update progress width

        // Call the function again to continue updating
        requestAnimationFrame(updateProgress);
      } else {
        // Once progress reaches 100%, stop the loop and do the following
        console.log("Progress reached 100!");

        // Mark task as completed
        taskContainer.classList.add('completed');
        statusSign.textContent = 'Completed'; // Change status to 'Completed'

        // Add custom class for completed status
        statusSign.classList.add('status-sign-end');

        // Change progress bar to gray when finished
        
        coffeeBreakProgressBar.classList.remove('blue-bar');
        coffeeBreakProgressBar.classList.add('gray-bar');

        // Log that the interval or loop was stopped
        console.log("Loop stopped and progress is completed.");
      }
    }

    // Start the loop
    requestAnimationFrame(updateProgress);
  }
});
